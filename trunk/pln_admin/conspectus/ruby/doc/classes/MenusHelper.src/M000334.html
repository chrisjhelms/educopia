  def archives_menu(params) 
    html = "UNDEFINED"; 
    if (params[:controller] != 'archives') then 
      html =  content_tag('div', 
         link_to('archives', :controller => :archives, :action => :list)); 
    else 
      html =  content_tag('div', 'archives');
      submenu = content_tag('div', 
            link_to('list-all', :controller => 'archives', :action => :list));
      if defined?(@archive) and !@archive.new_record? then 
        submenu =  submenu + 
          content_tag('div', 
            link_to('collections', :controller => 'archives', :action => :show)) +
          content_tag('div', 
           link_to('archival_units', :controller => 'archives', :action =>  :archival_units)) +
          content_tag('div', 
            link_to('plugins', :controller => 'archives', :action => :plugins)) +
          content_tag('div', 
           link_to('edit', :controller => 'archives', :action => :edit)) +
          content_tag('div', 
           link_to('configure', :configure => 'archives', :action => :configure)); 
     end
     submenu = content_tag('div', submenu, :class => 'indent')
     html = html + submenu;
     end 
    return html; 
  end
