  def content_provider_delete_link(cp) 
    if (is_super_user) then
      return link_to('delete',  
                     { :controller => :content_providers, 
                       :action => :destroy, 
                       :id => cp}, 
                 :method => :delete, 
                 :confirm => "Are you sure ?");
    else
      return ""; 
    end 
  end
