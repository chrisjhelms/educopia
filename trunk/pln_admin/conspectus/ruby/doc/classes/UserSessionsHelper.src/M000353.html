   def authentication_info() 
    if (current_user) then 
      provider = ""; 
      provider = "#{image_tag(current_user.content_provider.icon_url)}" if (current_user.content_provider); 
      rights = ""; 
      rights = content_tag(:span, "[SUPERUSER]", :class => "highlight")  if (current_user.rights == "super")
      return "#{provider} Welcome #{current_user.login} #{rights} | " + 
              link_to("My Profile", user_path(current_user)) + " | "  +
              link_to("Logout", {:controller => :user_sessions, 
                                 :action => :destroy}, :method => :delete)
    else 
      return link_to  "Login", new_user_session_url
    end
  end
