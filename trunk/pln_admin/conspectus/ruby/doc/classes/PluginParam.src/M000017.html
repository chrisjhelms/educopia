  def validate
   if (errors.empty?) then 
     if (plugin.nil?) then 
       errors.add_to_base("Plugin may not be nil"); 
     elsif (plugin.new_record?) then 
         errors.add_to_base  "Can't attach to unsaved plugin #{plugin.name}" 
     end 
   end
  end
