  def instance_table_start(headers, cls = "instance_table")
    str = "\n<table class=\'#{cls}\'>"; 
    if (!headers.nil? && !headers.empty?) then 
      th = ""; 
      headers.each { |h| th = th + content_tag(:th, h) } 
      str = str + content_tag(:thead, content_tag(:tr, th));  
    end 
    return str + "<tbody>"
  end
