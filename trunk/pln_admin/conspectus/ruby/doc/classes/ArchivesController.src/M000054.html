  def archival_units
    @archive = Archive.find(params[:id])
    @collections = @archive.collections; 
    @states =  params['states']
    if (@states.nil?) then  
      @state_name_list = nil; 
      @states = nil;
    else 
      @states = AuState.getList(params['states']); 
    end
    respond_to do |format|
      format.html # archival_units.html.erb
      format.xml  { render :xml =>  @archive.to_xml(:states => @states, :skip_instruct => true) }
    end
  end
