    def may_do(obj, user, action) 
      allow = true; 
      begin 
        allow = obj.send(:may_do_action, user, action); 
      rescue 
        allow = ApplicationController.may_do_action(obj, user, action);
      end
      if (!allow) then 
        flash[:error] = "You do not have access rights"
        redirect_to "/";
      end
      return allow; 
    end
