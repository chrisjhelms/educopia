  def test_get 
    prov = ContentProvider.find(:first); 
    assert(prov)
    p = Plugin.get(" #{prov.reverse_dns}.plugin  ", prov); 
    assert(p); 
    p2 = Plugin.get("#{prov.reverse_dns}.plugin", prov); 
    assert(p2)
    assert p == p2
    p3 = Plugin.get("   #{prov.reverse_dns}.plugin", prov); 
    assert(p3)
    assert p == p3
  end
