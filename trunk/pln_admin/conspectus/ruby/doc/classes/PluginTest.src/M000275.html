  def test_plugin_destroys 
    plugn = plugin(); 
    p = PluginParam.new(:plugin => plugn, 
                        :name => "x", :kind => "string"); 
    assert p.save
    
    p = PluginParam.new(:plugin => plugn, 
                        :name => "y", :kind => "string"); 
    assert p.save
    
    pid = plugn.id;
    plugn.destroy();
    assert(PluginParam.find(:all, :conditions  => { :plugin_id => pid }).count == 0); 
  end
