  def initialize(opts = nil) 
    if opts.key?(:param_values) && opts.key?(:collection) then
      param_str = "base_url=#{opts[:collection].base_url}"; 
      if (!opts[:param_values].empty?) then  
        param_str = [param_str, ArchivalUnit.hashToParamString(opts[:param_values])].join(PARAM_SPLIT)
      end 
      #puts "param_str #{param_str}"
      opts[:param_values] = param_str
      #puts "#{self.class.name}.initialize #{opts.inspect}"
    end  
    super(opts)
  end
