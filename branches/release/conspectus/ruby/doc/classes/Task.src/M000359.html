  def start_task(t, opts = {})
    task = t.name; 
    puts ">> ------------------------"
    puts ">> #{t.name} #{Time.now}"
    puts ">> ------------------------"
    @ret = 0; 
    @opts = RakeOption.new(t.name, opts);
    @dbase = Database.new(@opts['RAILS_ENV']); 
    @dry_run = @opts['DRY_RUN'] == "true";
    if (@dry_run) 
      @opts.info(" * "); 
    end 
    return !@dry_run
  end
