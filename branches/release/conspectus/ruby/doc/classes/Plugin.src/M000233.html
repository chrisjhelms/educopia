  def merge_params(hash)
    ret = true;
    hash.keys.each  { |pname|
      pp = param(pname)
      if (!pp.nil?) then
        pp[:kind] = hash[pname][:kind]
        pp[:descr] = hash[pname][:descr]
      else
        pp =  PluginParam.new(:name => pname,
                          :kind => hash[pname][:kind],
                          :descr => hash[pname][:descr],
                          :plugin => self)
      end
      ret = pp.save || ret;
    }
    return ret
  end
