  def self.find_by_plugin_baseurl(plgn, burl) 
    p = Plugin.find_by_name(plgn); 
    if (p.nil?) then
       return []; 
    end 
    col = Collection.find(:all, 
                          :conditions => { :base_url => burl, :plugin_id => p.id}); 
    return col;
  end
