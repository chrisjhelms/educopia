  def allow_transition_to(st2, with_super_power) 
    diff = self.level - st2.level; 
    # nobody should move from a state to another one more than one level away 
    if (diff < -1 || diff > 1) then return false; end
    if (diff == 0) then  return true; end
    # the super_user can move between any states separated by at most one level 
    if (with_super_power) then return true; end 
    if (st2.irreversible? || self.irreversible?) then  return false; end
    return true; 
  end
