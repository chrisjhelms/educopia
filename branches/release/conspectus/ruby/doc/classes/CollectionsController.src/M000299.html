  def list_mine
    if (current_user.content_provider.nil?) then 
      redirect_to :action => :list;
      return;
    end

    @collections = current_user.content_provider.collections(true);
    page = params[:page] || 1
    #@collections = Collection.paginate_by_content_provider current_user.content_provider.id, :page => page, :order => 'title'
    respond_to do |format|
      format.html { render :template => '/collections/list' }
      format.xml  { render :xml => @collections }
    end
  end
