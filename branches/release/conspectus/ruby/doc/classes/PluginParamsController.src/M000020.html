  def authenticate_edit_plugin() 
    @plugin = Plugin.find(params["plugin_id"]) if params.key?("plugin_id")
    @plugin_params= @plugin.plugin_params
    if (authenticate_edit_object(@plugin)) then  
      if (@plugin.has_aus) then 
        respond_to do |format|
          format.html { redirect_to(:id => @plugin.id, :controller => :plugins, :action => :show) }
          format.xml  { render :xml => @plugin_param.errors, :status => :unprocessable_entity }
        end 
      end
    end
  end
