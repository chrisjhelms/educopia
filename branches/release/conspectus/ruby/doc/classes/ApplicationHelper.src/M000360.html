  def select_obj_options(objs, method, selected, sort) 
    raise "objs empty" if objs.empty?
    string = "";
    if (sort) then 
      sorted =  objs.sort! { |a,b|  a.send(method) <=>  b.send(method)  } 
      objs = sorted; 
    end 
    objs.each { |o|
      string += "<option "
      if (o == selected) then 
        string += "selected ";
      end
      string += "> " + o.send(method); 
      string += "</option>\n";
    }
    return string;
  end
