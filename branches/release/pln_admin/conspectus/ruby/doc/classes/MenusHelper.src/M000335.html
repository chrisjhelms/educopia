  def content_providers_menu(params) 
    html = "UNDEFINED"; 
    if (params[:controller] != 'content_providers') then 
      html =  content_tag('div', 
      link_to('content_providers', :controller => :content_providers, :action => :list)); 
    else 
      html =  content_tag('div', 'content_providers');
      submenu = content_tag('div', 
      link_to('list-all', :controller => 'content_providers', :action => :list));
      if defined?(@content_provider) and !@content_provider.new_record? then 
        submenu =  submenu + 
        content_tag('div', 
        link_to('collections', :controller => 'content_providers', :action => :show)) +
        content_tag('div', 
        link_to('archival_units', :controller => 'content_providers', :action =>  :archival_units)) +
        content_tag('div', 
        link_to('plugins', :controller => 'content_providers', :action => :plugins)) +
        content_tag('div', 
        link_to('edit', :controller => 'content_providers', :action => :edit)); 
      end
      if (!current_user.nil? && current_user.super?) then 
         submenu =  submenu + 
          content_tag('div', 
            link_to('new', :controller => 'content_providers', :action => :new));
      end
      submenu = content_tag('div', submenu, :class => 'indent')
      html = html + submenu;
    end 
    return html; 
  end
