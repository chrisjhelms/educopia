  def menu(params) 
    html = "MENU"; 
    html =  archives_menu(params) + 
           content_providers_menu(params) + 
           collections_menu(params) + 
           plugins_menu(params) + 
           "";
    if (!current_user.nil? && current_user.super?) then 
      html += content_tag('div', "---") +  
              content_tag('div', users_menu(params)); 
    end
    return content_tag('div', html, :class => 'menu');
  end
