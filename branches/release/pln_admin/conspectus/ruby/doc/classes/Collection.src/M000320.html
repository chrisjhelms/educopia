  def metadata_url(action) 
    url = nil; 
    get_remote_metadata(); 
    if (remote_id) then 
      url = archive.metadata_url(action);
      if (url.nil? || url.empty?) then 
        return nil;
      end 
      url = url.gsub(/<TITLE>/, title);
      url = url.gsub(/<ID>/, "#{remote_id}");
    end
    return url;
  end
