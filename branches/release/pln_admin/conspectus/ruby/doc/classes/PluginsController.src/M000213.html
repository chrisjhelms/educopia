  def list_mine
    if (current_user.content_provider.nil?) then 
      redirect_to :action => :list;
      return;
    end

    @plugins = current_user.content_provider.plugins.select{ |p| !p.placeholder?}; 
    respond_to do |format|
      format.html { render :template => 'plugins/list' }
      format.xml  { render :xml => @plugins }
    end
  end
