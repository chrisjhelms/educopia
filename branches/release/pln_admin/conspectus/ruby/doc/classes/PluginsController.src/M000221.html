  def destroy
     respond_to do |format|    
      begin 
        @plugin.destroy
        format.html { redirect_to(plugins_url) }
        format.xml  { head :ok }
      rescue
        flash[:error] = "Can't destroy"
        format.html { redirect_to :action => "show" } 
        format.xml  { render :xml => @plugin.errors, :status => :unprocessable_entity }
      end  
    end
  end
