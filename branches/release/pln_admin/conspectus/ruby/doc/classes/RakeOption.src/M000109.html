   def initialize(name, hash)
      if (ENV.key?('BUNDLE')) 
        bundle = Opts::GLOBALS.fetch('RAKE_BUNDLES',{})[ENV['BUNDLE']];
        hash = hash.merge(Opts::GLOBALS.fetch('RAKE_BUNDLES',{})[ENV['BUNDLE']])
      end
      super(name, DEFAULT_OPTS.merge(hash))
      self.merge(ENV);
      ALL_OPTS[name] = self; 
   end
