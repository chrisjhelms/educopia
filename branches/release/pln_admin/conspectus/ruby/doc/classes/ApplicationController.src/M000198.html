    def ApplicationController.may_do_action(obj, user, action) 
      return true if (:action == :read); 
      return true if (!user.nil? && user.rights == "super");
      # if the thing does not have a content_provider user can not edit 
      # that is if there is an exception on content_provider access return false;
      begin
        cp = obj.content_provider
        return (cp == user.content_provider) && (user.rights == "edit");
      rescue
        return false 
      end
    end
