  def self.get(name, contentProv) 
    name = ActiveRecordExtension.compact_string(name)
    p = find_by_name(name)
    if (!p.nil?) then 
        return nil if  p.content_provider != contentProv
        return p
    end
    p = Plugin.new(:name => name, :content_provider => contentProv);
    if (!p.save) then 
      return nil; 
    end
    return p;
  end
