  def test_no_double_name 
    c = ContentProvider.new(:name => "University Somewhere edu", 
    :reverse_dns => "edu.uni", 
    :home_url => "http://uni.edu"); 
    assert c.save!
    
    # test no double names
    c = ContentProvider.new(:name => "  University   Somewhere   edu", 
    :reverse_dns => "edu.uni.lib", 
    :home_url => "http://library.uni.edu"); 
    assert !c.save
  end
