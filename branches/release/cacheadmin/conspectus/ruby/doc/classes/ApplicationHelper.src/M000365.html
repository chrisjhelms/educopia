  def span_short(str, len = 40)
     str ||= ''
     shorten = (str.length > len)
     if (shorten) then 
       shortvalue = (shorten ? "#{str[0..(len-1)]}..." : str)
       return "<span title=\'#{str}\'> " + shortvalue + "</span>"
     else
       return "<span> " + str + "</span>"
     end
  end
