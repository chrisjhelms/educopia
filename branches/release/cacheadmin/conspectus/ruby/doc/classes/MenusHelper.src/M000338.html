  def users_menu(params) 
    html = "UNDEFINED"; 
    if (params[:controller] != 'users' && params[:controller] != 'user_sessions') then
       html =  content_tag('div', 
         link_to('users', :controller => :users, :action => :index)); 
    else 
      html =  content_tag('div', 'users');
      submenu = 
         content_tag('div', 
            link_to('list', :controller => :users, :action => :index)) +
         content_tag('div', 
            link_to('new', :controller => :users, :action => :new)) +
         content_tag('div', 
            link_to('check_pwd', :controller => :user_sessions, :action => :new_guess_pwd));
       submenu = content_tag('div', submenu, :class => 'indent')
       html = html + submenu;
    end 
    return html; 
  end
