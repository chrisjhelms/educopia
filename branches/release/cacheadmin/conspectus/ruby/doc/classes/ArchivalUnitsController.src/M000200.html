  def new
    @archival_unit = ArchivalUnit.new(:collection => @collection, 
                                      :param_values => {})
    respond_to do |format|
      if (@collection.plugin.plugin_params.length == 1) then 
        if @archival_unit.save
          flash[:notice] = 'ArchivalUnit was successfully created.'
          format.html { redirect_to collection_url(@collection) }
          format.xml  { render :xml => @archival_unit, :status => :created, :location => @archival_units }
        else
          flash[:error] = 'Could not create ArchivalUnit.'
          format.html { render :action => "new" }
          format.xml  { render :xml => @archival_unit.errors, :status => :unprocessable_entity }
        end 
      else
        format.html # new.html.erb
        format.xml  { render :xml => @archival_units }
      end 
    end
  end
