  def to_xml(options={})
       options[:indent] ||= 2
       xml = options[:builder] ||= Builder::XmlMarkup.new(:indent => options[:indent])

       xml.instruct! unless options.delete(:skip_instruct)     
       archival_units.each do |au| 
          au.to_xml(:builder => xml, :skip_instruct => true, :collection => self)               
       end
       return xml
   end
