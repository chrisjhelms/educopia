  def plugin_prefix=(str) 
    cstr = compact_string(str)
    if (plugins.count > 0 and self[:plugin_prefix] != cstr) then  
        errors.add('plugin_prefix',  "#{self.class.name} #{name} has plugins; can't change plugin_prefix") 
    end 
    self[:plugin_prefix] = cstr
  end
