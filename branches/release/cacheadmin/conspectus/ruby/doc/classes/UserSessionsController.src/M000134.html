  def guess_pwd
    @user_session = UserSession.new()
    @pwd = params[:user_session][:password];
    @users = []; 
    for u in User.find(:all) do 
      session = UserSession.new( :login => u.login, :password => @pwd); 
      if (session.valid?) then 
        @users << u; 
      end
    end
  end
