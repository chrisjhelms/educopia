  def self.get_xml_from_file(file) 
    name = ""; 
    if (file == "") then
      raise"must give FILE parameter"
    end 
    name = file.match('(^.*).xml')
    if (!name) then 
      raise  "#{file} does not end in .xml"; 
    end   
    xml = "EMPTY"; 
    begin 
      xml = File.read(file); 
    rescue 
      raise  "Can't read #{file}";
    end
    path = name[1].split(File::SEPARATOR)
    return [path, xml]; 
  end
