 def self.to_csv(options={})
       options[:separator] ||= '\t';
       options[:au_list] ||= []
       aus = options[:au_list]
       sep = options[:separator]
       str = ""
       aus.each do  |au| 
          str +=  "#{au.collection.title}#{sep}"  + 
               "#{au.collection.plugin.name}#{sep}" +
               "#{au.collection.base_url}#{sep}" +               
               "#{au.param_values}\n";
       end 
       return str;
 end
